class Node(object):
    def __init__(self, data=None, next=None):
        self.data = data
        self.next = next


class LinkedList(object):
    def __init__(self, head=None):
        self.head = head

    def insert(self, data):
        new_node = Node(data)
        new_node.next(self.head)
        self.head = new_node

    def search(self, data):
        current = self.head
        previous = None
        found = False

        while current and found is False:
            if current.data == data:
                found = True
            else:
                previous = current
                current = current.next

        if current is None:
            return False
        else:
            return {
                'c': current,
                'p': previous
            }

    def delete(self, data):
        found = self.search(data)

        if found:
            found['p'].next = found['c'].next

    def size(self):
        current = self.head
        count = 0

        while current:
            count += 1
            current = current.get_next()

        return count
